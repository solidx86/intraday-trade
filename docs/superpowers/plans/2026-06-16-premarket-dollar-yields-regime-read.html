    found = [t for t in REGIME_TOKENS if t in body]
    if found:
        return CheckResult("regime_read", True, f"regime={found[0]}")
    return CheckResult(
        "regime_read",
        False,
        "no SCARED/GREEDY/GOLDILOCKS/NEUTRAL regime token in section 1.1",
